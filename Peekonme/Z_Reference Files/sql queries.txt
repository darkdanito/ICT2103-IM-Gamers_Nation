create view avg_rating as
select r1.*, avg(r1.rating) as avgrating from review_have r1, review_have r2 where (r1.GameID = r2.GameID and r1.userid <> r2.userid) or (r1.GameID = r2.GameID and r1.userid = r2.userid)
